class Solution {  
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {  
        ListNode p1 = l1, p2 = l2;  
        ListNode dummy = new ListNode(-1);  
        ListNode p = dummy;  
        //进位，(比如8+5=13,carry就是1)  
        int carry = 0;  
        while (p1 != null || p2 != null || carry > 0){  
            //先把上一轮的进位拿过来  
            int val = carry;  
            if (p1 != null){  
                val += p1.val;  
                p1 = p1.next;  
            }  
            if (p2 != null){  
                val += p2.val;  
                p2 = p2.next;  
            }  
            //此时val是两数之和  
            carry = val /10;  
            val = val % 10;  
            p.next = new ListNode(val);  
            //指针后移  
            p = p.next;  
        }  
        return dummy.next;  
    }  
}
