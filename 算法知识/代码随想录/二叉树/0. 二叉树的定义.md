```java
public class TreeNode {
    // 1. 数据域：存数值
    public int val;

    // 2. 指针域：左孩子 (Left Child)
    public TreeNode left;

    // 3. 指针域：右孩子 (Right Child)
    public TreeNode right;

    // --- 构造方法 (Constructors) ---
    
    // 无参构造
    public TreeNode() {}

    // 只有数值的构造 (最常用)
    public TreeNode(int val) { 
        this.val = val; 
    }

    // 全参构造
    public TreeNode(int val, TreeNode left, TreeNode right) {
        this.val = val;
        this.left = left;
        this.right = right;
    }
}
```
