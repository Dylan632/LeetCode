```java
public class ListNode {
    // 1. 数据域：这里存的是具体的值
    public int val;

    // 2. 指针域：存的是“下一个节点对象的地址”
    // 注意：它的类型也是 ListNode！这就是“我包含我自己”的定义。
    public ListNode next;

    // --- 构造方法 (用来创建新节点的) ---
    
    // 无参构造
    public ListNode() {}

    // 最常用的构造：创建一个带数值的节点
    public ListNode(int val) { 
        this.val = val; 
    }

    // 全参构造：既有值，又直接指好了下一个是谁
    public ListNode(int val, ListNode next) { 
        this.val = val; 
        this.next = next; 
    }
}
```
