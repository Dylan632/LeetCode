class Solution {  
    public int longestConsecutive(int[] nums) {  
        if (nums.length == 0) {  
            return 0;  
        }  

        Set<Integer> numSet = new HashSet<>();  
        for (int num : nums) {  
            numSet.add(num);  
        }  

        int res = 0;  
        for (int start : numSet) {  
            // å¦‚æœåŒ…å«(start-1)ï¼Œè¯´æ˜startä¸æ˜¯èµ·ç‚¹ï¼Œè·³è¿‡  
            if (numSet.contains(start - 1)) {  
                continue;  
            }  
            int next = start + 1;  
            while (numSet.contains(next)) {  
                next++;  
            }  
            res = Math.max(res, next - start);  
        }  
        return res;  
    }  
}
